<body>
    <h1>Calculadora Básica</h1>
    <input type="number" id="numero1" placeholder="Número 1">
    <input type="number" id="numero2" placeholder="Número 2">
    <br>
    <button id="sumarBtn">Sumar</button>
    <button id="restarBtn">Restar</button>
    <button id="multiplicarBtn">Multiplicar</button>
    <button id="dividirBtn">Dividir</button>

    <div id="resultado"></div>

    <script>
        const input1 = document.getElementById('numero1');
        const input2 = document.getElementById('numero2');
        const resultadoDiv = document.getElementById('resultado');

        function obtenerNumeros() {
            const num1 = parseFloat(input1.value);
            const num2 = parseFloat(input2.value);
            return { num1, num2 };
        }

        document.getElementById('sumarBtn').addEventListener('click', () => {
            const { num1, num2 } = obtenerNumeros();
            resultadoDiv.textContent = !isNaN(num1) && !isNaN(num2)
                ? `Resultado: ${num1 + num2}`
                : 'Ingresa dos números válidos.';
        });

        document.getElementById('restarBtn').addEventListener('click', () => {
            const { num1, num2 } = obtenerNumeros();
            resultadoDiv.textContent = !isNaN(num1) && !isNaN(num2)
                ? `Resultado: ${num1 - num2}`
                : 'Ingresa dos números válidos.';
        });

        document.getElementById('multiplicarBtn').addEventListener('click', () => {
            const { num1, num2 } = obtenerNumeros();
            resultadoDiv.textContent = !isNaN(num1) && !isNaN(num2)
                ? `Resultado: ${num1 * num2}`
                : 'Ingresa dos números válidos.';
        });

        document.getElementById('dividirBtn').addEventListener('click', () => {
            const { num1, num2 } = obtenerNumeros();
            if (!isNaN(num1) && !isNaN(num2)) {
                if (num2 === 0) {
                    resultadoDiv.textContent = 'No se puede dividir por cero.';
                } else {
                    resultadoDiv.textContent = `Resultado: ${num1 / num2}`;
                }
            } else {
                resultadoDiv.textContent = 'Ingresa dos números válidos.';
            }
        });
    </script>
