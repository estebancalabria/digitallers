    <input type="text" id="nombreInput" placeholder="Tu nombre">
    <br>
    <button id="saludarBtn">Saludar</button>

    <script>
        const button = document.getElementById('saludarBtn');
        const input = document.getElementById('nombreInput');

        button.addEventListener('click', () => {
            const nombre = input.value.trim();
            if (nombre) {
                alert(`¡Hola, ${nombre}! ¿Cómo estás?`);
            } else {
                alert('Por favor, escribe tu nombre primero.');
            }
        });
    </script>
